function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); if (!sheet) { return jsonOutput({ success: false, message: "Sheet1 tidak ditemukan" }); } var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.nama || "", data.tanggal || "", Number(data.pemasukan || 0), Number(data.pengeluaran || 0), data.kategori || "", new Date() ]); return jsonOutput({ success: true, message: "Data berhasil disimpan" }); } catch (err) { return jsonOutput({ success: false, message: err.toString() }); } } function doGet(e) { try { var action = e.parameter.action || ""; var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); if (!sheet) { return jsonOutput({ success: false, message: "Sheet1 tidak ditemukan" }); } if (action === "list") { var user = (e.parameter.user || "").toLowerCase().trim(); var lastRow = sheet.getLastRow(); var lastCol = sheet.getLastColumn(); if (lastRow < 2) { return jsonOutput({ success: true, data: [] }); } var values = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues(); var result = []; values.forEach(function(row) { var nama = String(row[0] || "").toLowerCase().trim(); var tanggal = row[1]; var pemasukan = Number(row[2] || 0); var pengeluaran = Number(row[3] || 0); var kategori = row[4] || ""; if (user === "admin" || user === nama) { result.push({ nama: nama, tanggal: formatTanggal(tanggal), pemasukan: pemasukan, pengeluaran: pengeluaran, kategori: kategori }); } }); return jsonOutput({ success: true, data: result }); } return jsonOutput({ success: true, message: "FINTRACK API READY" }); } catch (err) { return jsonOutput({ success: false, message: err.toString() }); } } function formatTanggal(tanggal) { if (Object.prototype.toString.call(tanggal) === "[object Date]" && !isNaN(tanggal)) { return Utilities.formatDate(tanggal, Session.getScriptTimeZone(), "yyyy-MM-dd"); } return tanggal ? String(tanggal) : ""; } function jsonOutput(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
